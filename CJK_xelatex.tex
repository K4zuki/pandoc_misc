\def\surplus#1#2{%
        \newcount\n \newcount\m \newcount\t
        \n=#1 \m=#2 \t=\n%
        \divide \n by \m%
        \multiply \n by \m%
        \advance \t by -\n%
        #1=\t\relax%
}
